    VAR
      Text000@1000 : TextConst 'ENU=You must not use a "%1" %2 when %3 is "%4".;ESP=No debe usar un"%1" %2 cuando %3 es "%4".';

      Text003@1003 : TextConst 'ENU=%1 %2 must not be mentioned.;ESP=%1 %2 no debe ser mencionado.';
      Text004@1004 : TextConst 'ENU=%1 %2 must be %3.;ESP=%1 %2debe ser %3.';

    PROCEDURE CheckAndMakeErrorMessage@1(DefaultDim1@1000 : Record 352;DefaultDim2@1001 : Record 352;VAR ErrorMessage@1002 : Text[250]);
    BEGIN
      CASE DefaultDim1."Value Posting" OF

        DefaultDim1."Value Posting"::" ":
          IF (((DefaultDim2."Value Posting" = DefaultDim2."Value Posting"::"Same Code") AND
               (DefaultDim2."Dimension Value Code" = '')) OR
               (DefaultDim2."Value Posting" = DefaultDim2."Value Posting"::"No Code")) AND
               (DefaultDim1."Dimension Value Code" <> '') THEN
