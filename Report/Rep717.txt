      Supply@1003 : Decimal;
      ProjectedInventory@1004 : Decimal;

      NewItem.SETRANGE("Date Filter",0D,31129999D);
      NewItem.CALCFIELDS(Inventory);
      Demand :=
        AvailToPromise.CalcGrossRequirement(NewItem) +
        AvailToPromise.CalcOutboundReservations(NewItem);
      Supply :=
        AvailToPromise.CalcScheduledReceipt(NewItem) +
        AvailToPromise.CalcInboundReservations(NewItem);

        ELSE
          ProjectedInventory := QtyAvailable;
      END;
      ReorderQty :=
        AvailabilityMgt.GetItemReorderQty(NewItem,ProjectedInventory);
    END;
