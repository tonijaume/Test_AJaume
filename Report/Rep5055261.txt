    LOCAL PROCEDURE CalcLine@1140000(Level@1140000 : Integer) : Boolean;
    VAR
      PostingType@1140001 : 'Standard,Heading,Total,Begin-Total,End-Total';
      ThisPostingType@1140002 : 'Standard,Heading,Total,Begin-Total,End-Total';
      ThisDimValCode@1140003 : Code[20];
      Totaling@1140004 : Text[250];
      ThisDimValName@1140005 : Text[50];
      ThisTotaling@1140006 : Text[250];
      i@1140007 : Integer;
      Indentation@1140008 : Integer;
      ThisIndentation@1140009 : Integer;
      LineNo@1140010 : Integer;
      LowestLevel@1140011 : Boolean;
      HasValue@1140012 : Boolean;
      More@1140013 : Boolean;
    BEGIN
      FOR i := Level + 1 TO 4 DO
        SetAccSchedLineFilter(DimCode[i],'',FALSE,'');

      IF Iteration(
        FindFirstDim[Level],DimCode[Level],DimValCode[Level],DimValName[Level],LevelFilter[Level],
        Totaling,Indentation,PostingType)
      THEN BEGIN
        IF Level = 4 THEN
          LowestLevel := TRUE
        ELSE
          LowestLevel :=  DimCode[Level + 1] = '';

        IF (NOT PrintEmptyLines) AND (NOT LowestLevel) THEN BEGIN
          SetAccSchedLineFilter(DimCode[Level],DimValCode[Level],TRUE,Totaling);
          HasValue := TestCalcLine(Level + 1,TRUE);
          FOR i := Level + 1 TO 4 DO
            SetAccSchedLineFilter(DimCode[i],'',FALSE,'');
        END;

        ShowBold[Level] := NOT LowestLevel OR (PostingType <> PostingType::Standard);

        IF LowestLevel THEN
          DimValNameIndent[Level] := Indentation
        ELSE BEGIN
          DimValNameIndent[Level] := 0;
          CLEAR(PrintEndTotals);
        END;

        // Check if begin-total should be printed...
        IF (NOT PrintEmptyLines) AND LowestLevel AND (PostingType = PostingType::"Begin-Total") THEN BEGIN
          LineNo := AccSchedLine."Line No.";
          ThisDimValCode := DimValCode[Level];
          ThisTotaling := Totaling;
          ThisIndentation := 999999999;
          More := TRUE;
          SetAccSchedLineFilter(DimCode[Level],ThisDimValCode,TRUE,ThisTotaling);
          HasValue := CalcColumns(0);
          WHILE More AND (NOT HasValue) AND (ThisIndentation > Indentation) DO BEGIN
            More :=
              Iteration(
                FindFirstDim[Level],DimCode[Level],ThisDimValCode,ThisDimValName,LevelFilter[Level],
                ThisTotaling,ThisIndentation,ThisPostingType);
            IF More THEN BEGIN
              SetAccSchedLineFilter(DimCode[Level],ThisDimValCode,TRUE,ThisTotaling);
              HasValue := CalcColumns(0);
            END;
          END;
          AccSchedLine."Line No." := LineNo;
          PrintEndTotals[Indentation + 1] := HasValue;
        END;

        // Check if end-total should be printed...
        IF (NOT PrintEmptyLines) AND LowestLevel AND (PostingType = PostingType::"End-Total") THEN BEGIN
          HasValue := PrintEndTotals[Indentation + 1];
          PrintEndTotals[Indentation + 1] := FALSE;
        END;

        SetAccSchedLineFilter(DimCode[Level],DimValCode[Level],TRUE,Totaling);
        FOR i := 1 TO MaxColumnsDisplayed DO BEGIN
          ColumnValuesDisplayed[i] := 0;
          ColumnValuesAsText[i,Level] := '';
        END;

        EXIT(HasValue OR CalcColumns(Level));
      END ELSE
        CurrReport.BREAK;
    END;

    LOCAL PROCEDURE TestCalcLine@1140001(Level@1140000 : Integer;ThisFindFirstDim@1140001 : Boolean) : Boolean;
    VAR
      Totaling@1140002 : Text[250];
      ThisDimValName@1140003 : Text[50];
      ThisIndentation@1140004 : Integer;
      ThisPostingType@1140005 : 'Standard,Heading,Total,Begin-Total,End-Total';
      LowestLevel@1140006 : Boolean;
      HasValue@1140007 : Boolean;
      More@1140008 : Boolean;
    BEGIN
      FindFirstDim[Level] := ThisFindFirstDim;

      IF Iteration(
           FindFirstDim[Level],DimCode[Level],DimValCode[Level],ThisDimValName,LevelFilter[Level],
           Totaling,ThisIndentation,ThisPostingType)
      THEN BEGIN
        IF Level = 4 THEN
          LowestLevel := TRUE
        ELSE
          LowestLevel :=  DimCode[Level + 1] = '';

        IF LowestLevel THEN BEGIN
          More := TRUE;
          SetAccSchedLineFilter(DimCode[Level],DimValCode[Level],TRUE,Totaling);
          HasValue := CalcColumns(0);
          WHILE More AND (NOT HasValue) DO BEGIN
            More :=
              Iteration(
                FindFirstDim[Level],DimCode[Level],DimValCode[Level],ThisDimValName,LevelFilter[Level],
                Totaling,ThisIndentation,ThisPostingType);
            IF More THEN BEGIN
              SetAccSchedLineFilter(DimCode[Level],DimValCode[Level],TRUE,Totaling);
              HasValue := CalcColumns(0);
            END;
          END;
        END ELSE BEGIN
          HasValue := TestCalcLine(Level + 1,TRUE);
          IF NOT HasValue THEN
            HasValue := TestCalcLine(Level,FindFirstDim[Level]);
        END;
      END ELSE
        HasValue := FALSE;

      EXIT(HasValue);
    END;

    LOCAL PROCEDURE CalcColumns@1140002(Level@1140000 : Integer) : Boolean;
    VAR
      NonZero@1140001 : Boolean;
      i@1140002 : Integer;
    BEGIN
      NonZero := FALSE;
      IF NOT LiqAccFilterSet THEN
        AccSchedLine.Totaling := LiqAccRange;

      WITH ColLayoutTmp DO BEGIN
        SETRANGE("Column Layout Name",ColumnLayoutName);
        i := 0;
        IF FIND('-') THEN
          REPEAT
            IF ColLayoutTmp.Show <> ColLayoutTmp.Show::Never THEN BEGIN
              i := i + 1;
              AccSchedLine."Line No." := AccSchedLine."Line No." + 1;
              ColumnValuesDisplayed[i] :=
                AccSchedManagement.CalcCell(AccSchedLine,ColLayoutTmp);
              NonZero :=
                NonZero OR (ColumnValuesDisplayed[i] <> 0) AND
                (ColLayoutTmp."Column Type" <> ColLayoutTmp."Column Type"::Formula);
              IF Level > 0 THEN
                ColumnValuesAsText[i,Level] :=
                  AccSchedManagement.FormatCellAsText(ColLayoutTmp,ColumnValuesDisplayed[i]);
            END;
          UNTIL (i >= MaxColumnsDisplayed) OR (NEXT = 0);
      END;
      EXIT(NonZero);
    END;

    LOCAL PROCEDURE Iteration@1140004(VAR FindFirst@1140000 : Boolean;IterationDimCode@1140001 : Text[30];VAR IterationDimValCode@1140002 : Code[20];VAR IterationDimValName@1140003 : Text[50];IterationFilter@1140004 : Text[250];VAR IterationTotaling@1140005 : Text[250];VAR IterationIndentation@1140006 : Integer;VAR IterationPostingType@1140007 : 'Standard,Heading,Total,Begin-Total,End-Total') : Boolean;
    VAR
      SearchResult@1140008 : Boolean;
    BEGIN
      CASE IterationDimCode OF
        TempLiqAcc.TABLECAPTION :
        BEGIN
          TempLiqAcc.RESET;
          TempLiqAcc.SETFILTER("No.",IterationFilter);
          IF FindFirst THEN
            SearchResult := TempLiqAcc.FIND('-')
          ELSE
            IF TempLiqAcc.GET(IterationDimValCode) THEN
              SearchResult := (TempLiqAcc.NEXT <> 0);
          IF SearchResult THEN BEGIN
            IterationDimValCode := TempLiqAcc."No.";
            IterationDimValName := TempLiqAcc.Name;
            IterationTotaling := TempLiqAcc.Totaling;
            IterationIndentation := TempLiqAcc.Indentation;
            IterationPostingType := TempLiqAcc."Account Type";
          END;
        END;
        TempLiquidity.TABLECAPTION :
        BEGIN
          TempLiquidity.RESET;
          TempLiquidity.SETFILTER("No.",IterationFilter);
          IF FindFirst THEN
            SearchResult := TempLiquidity.FIND('-')
          ELSE
            IF TempLiquidity.GET(IterationDimValCode) THEN
              SearchResult := (TempLiquidity.NEXT <> 0);
          IF SearchResult THEN BEGIN
            IterationDimValCode := TempLiquidity."No.";
            IF TempLiquidity."No." <> '' THEN
              IterationDimValName := TempLiquidity.Description
            ELSE
              IterationDimValName := Text1140009;
            IterationIndentation := 0;
            IterationPostingType := 0;
          END;
        END;
        ELSE BEGIN
          TempDimVal.RESET;
          TempDimVal.SETRANGE("Dimension Code",IterationDimCode);
          TempDimVal.SETFILTER(Code,IterationFilter);
          IF FindFirst THEN
            SearchResult := TempDimVal.FIND('-')
          ELSE
            IF TempDimVal.GET(IterationDimCode,IterationDimValCode) THEN
              SearchResult := (TempDimVal.NEXT <> 0);
          IF SearchResult THEN BEGIN
            IterationDimValCode := TempDimVal.Code;
            IterationDimValName := TempDimVal.Name;
            IterationTotaling := TempDimVal.Totaling;
            IterationIndentation := TempDimVal.Indentation;
            IterationPostingType := TempDimVal."Dimension Value Type";
          END;
        END;
      END;
      IF NOT SearchResult THEN BEGIN
        IterationDimValCode := '';
        IterationDimValName := '';
        IterationTotaling := '';
        IterationIndentation := 0;
        IterationPostingType := 0;
      END;
      FindFirst := FALSE;
      EXIT(SearchResult);
    END;

    LOCAL PROCEDURE SetAccSchedLineFilter@1140005(AnalysisViewDimCode@1140000 : Text[30];AnalysisViewFilter@1140001 : Text[250];SetFilter@1140002 : Boolean;Totaling@1140003 : Text[250]);

            ELSE
              AccSchedLine.Totaling := LiqAccRange;

        TempLiquidity.TABLECAPTION:
          IF SetFilter THEN
            AccSchedLine.SETFILTER("Liquidity Filter",AnalysisViewFilter)
          ELSE
            AccSchedLine.SETRANGE("Liquidity Filter");

        "Liquidity Analysis View"."Dimension 1 Code":
           IF SetFilter THEN
             AccSchedLine."Dimension 1 Totaling" := AnalysisViewFilter

             AccSchedLine."Dimension 1 Totaling" := '';

           IF SetFilter THEN
             AccSchedLine."Dimension 2 Totaling" := AnalysisViewFilter
           ELSE
             AccSchedLine."Dimension 2 Totaling" := '';

           IF SetFilter THEN
             AccSchedLine."Dimension 3 Totaling" := AnalysisViewFilter
           ELSE
             AccSchedLine."Dimension 3 Totaling" := '';
        "Liquidity Analysis View"."Dimension 4 Code":
           IF SetFilter THEN
             AccSchedLine."Dimension 4 Totaling" := AnalysisViewFilter
           ELSE
             AccSchedLine."Dimension 4 Totaling" := '';
      END;
    END;
