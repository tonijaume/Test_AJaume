    VAR
      Text000@1007 : TextConst 'ENU=It was not possible to find a %1 in %2.;ESP=No se ha podido encontrar un %1 en %2.';

      PeriodBeginBalance@1017 : Decimal;

      FiscalYearFilter@1014 : Text[250];
      CustFilter@1004 : Text[1024];
      PeriodStartDate@1003 : Date;
      PeriodEndDate@1002 : Date;
      FiscalYearStartDate@1001 : Date;
      Text003@1028 : TextConst 'ENU=Period: %1;ESP=Periodo: %1';
      Text004@1027 : TextConst 'ENU=Total for;ESP=Total para';
      Text005@1000 : TextConst 'ENU=Group Totals: %1;ESP=Totales grupo: %1';

    LOCAL PROCEDURE CalcAmounts@1(DateFrom@1005 : Date;DateTo@1004 : Date;VAR BeginBalance@1003 : Decimal;VAR DebitAmt@1002 : Decimal;VAR CreditAmt@1001 : Decimal;VAR TotalBalance@1000 : Decimal);
    VAR
      DtlCustLedgEntry@1007 : Record 379;
    BEGIN
      Customer.SETRANGE("Date Filter",0D,DateFrom - 1);
      Customer.CALCFIELDS("Net Change (LCY)");
      BeginBalance := Customer."Net Change (LCY)";

      WITH DtlCustLedgEntry DO BEGIN
        SETCURRENTKEY("Customer No.","Posting Date","Entry Type","Currency Code");
        SETRANGE("Customer No.",Customer."No.");
        SETRANGE("Posting Date",DateFrom,DateTo);
        SETFILTER("Entry Type",'<> %1',"Entry Type"::Application);
        CALCSUMS("Debit Amount (LCY)","Credit Amount (LCY)");
        DebitAmt := "Debit Amount (LCY)";
        CreditAmt := "Credit Amount (LCY)";
        TotalBalance := BeginBalance + DebitAmt - CreditAmt;
      END;
    END;

    BEGIN
    END.
