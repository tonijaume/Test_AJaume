    VAR
      i@1002 : Integer;

    END;

    LOCAL PROCEDURE CalcVariance@4();
    VAR
      i@1000 : Integer;
    BEGIN
      FOR i := 1 TO ARRAYLEN(VarPct) DO
        VarPct[i] := CalcIndicatorPct(ExpCost[i],ActCost[i]);
    END;

    LOCAL PROCEDURE CalcIndicatorPct@3(Value@1000 : Decimal;Sum@1001 : Decimal) : Decimal;
    BEGIN
      IF Value = 0 THEN
        EXIT(0);

      EXIT(ROUND((Sum - Value) / Value * 100,1));
    END;
