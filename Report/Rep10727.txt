    PROCEDURE CalcCloseEntriesHeading@5();
    VAR
      HistoricGLAcc@1100000 : Record 10721;
    BEGIN
      HistoricGLAcc.SETRANGE("Date Filter",0D,CLOSINGDATE(ToDate));
      IF GLFilterAccType = GLFilterAccType::Heading THEN BEGIN
        HistoricGLAcc.SETFILTER("No.","Historic G/L Account".Totaling);
        HistoricGLAcc.SETRANGE("Account Type",0);
      END ELSE
        HistoricGLAcc.SETFILTER("No.","Historic G/L Account"."No.");
      IF HistoricGLAcc.FIND('-') THEN BEGIN
        REPEAT
          HistoricGLAcc.CALCFIELDS("Additional-Currency Net Change","Net Change");
          IF PrintAmountsInAddCurrency THEN BEGIN
            IF HistoricGLAcc."Additional-Currency Net Change" > 0 THEN
              CreditAmt := CreditAmt + HistoricGLAcc."Additional-Currency Net Change"
            ELSE
              DebitAmt := DebitAmt + ABS(HistoricGLAcc."Additional-Currency Net Change");
          END ELSE BEGIN
            IF HistoricGLAcc."Net Change" > 0 THEN
              CreditAmt := CreditAmt + HistoricGLAcc."Net Change"
            ELSE
              DebitAmt := DebitAmt + ABS(HistoricGLAcc."Net Change");
          END;
        UNTIL HistoricGLAcc.NEXT = 0;
      END;
    END;

        HistoricGLAcc.SETFILTER("No.","Historic G/L Account".Totaling);
        HistoricGLAcc.SETRANGE("Account Type",0);
      END ELSE
        HistoricGLAcc.SETFILTER("No.","Historic G/L Account"."No.");
      IF HistoricGLAcc.FIND('-') THEN BEGIN
        REPEAT
          HistoricGLAcc.CALCFIELDS("Additional-Currency Net Change","Net Change");
          IF PrintAmountsInAddCurrency THEN BEGIN
            IF HistoricGLAcc."Additional-Currency Net Change" < 0 THEN
              CreditAmt := CreditAmt + ABS(HistoricGLAcc."Additional-Currency Net Change")
            ELSE
              DebitAmt := DebitAmt + HistoricGLAcc."Additional-Currency Net Change";
          END ELSE BEGIN
            IF HistoricGLAcc."Net Change" < 0 THEN
              CreditAmt := CreditAmt + ABS(HistoricGLAcc."Net Change")
            ELSE
              DebitAmt := DebitAmt + HistoricGLAcc."Net Change";
          END;
        UNTIL HistoricGLAcc.NEXT = 0;
      END;
    END;

    PROCEDURE Verify@8(Process@1100000 : Integer) : Integer;
    VAR
      AccPeriod@1100001 : Record 50;
      AccPeriod2@1100002 : Record 50;
    BEGIN
      CASE Process OF
        1:
          BEGIN
            AccPeriod.SETRANGE("New Fiscal Year",TRUE);
            AccPeriod.SETRANGE("Starting Date",FromDate);
            IF AccPeriod.FIND('-') THEN
              EXIT(1)
            ELSE
              EXIT(0);
          END;
        2:
          BEGIN
            AccPeriod.SETRANGE("New Fiscal Year",TRUE);
            AccPeriod.SETRANGE("Starting Date",ToDate + 1);
            IF AccPeriod.FIND('-') THEN
              EXIT(1)
            ELSE
              EXIT(0);
          END;
        3:
          BEGIN
            IF Open AND (FromDate = ToDate) THEN
              EXIT(0);
            AccPeriod.SETRANGE("Starting Date",FromDate,ToDate);
            IF AccPeriod.FIND('-') THEN
              EXIT(AccPeriod.COUNT)
            ELSE
              EXIT(0);
          END;
        4:
          BEGIN
            IF (FromDate <> ToDate) THEN BEGIN
              AccPeriod.SETRANGE("Starting Date",FromDate + 1,ToDate);
              AccPeriod.SETRANGE("New Fiscal Year",TRUE);
              IF AccPeriod.FIND('-') THEN BEGIN
                ERROR(Text1100005);
                CurrReport.BREAK;
              END;
            END;
          END;
        5:
          BEGIN
            AccPeriod.SETRANGE("Starting Date",FromDate);
            IF NOT AccPeriod.FIND('-') THEN
              ERROR(Text1100006,FromDate);
          END;
        6:
          BEGIN
            AccPeriod.SETRANGE("Starting Date",ToDate + 1);
            IF NOT AccPeriod.FIND('-') THEN
              ERROR(Text1100007);
          END;
        7:
          BEGIN
            AccPeriod2.SETFILTER("Starting Date",'<%1',FromDate);
            AccPeriod2.SETRANGE("New Fiscal Year",TRUE);
            IF AccPeriod2.FIND('+') THEN;
            AccPeriod.SETRANGE("Starting Date",AccPeriod2."Starting Date",FromDate);
            IF AccPeriod.FIND('-') THEN
              EXIT(AccPeriod.COUNT)
            ELSE
              EXIT(0);
          END;
        ELSE;
      END;
    END;

    BEGIN
    END.
