    LOCAL PROCEDURE CalcProfitPct@1();
    BEGIN
      WITH ValueEntryBuffer DO BEGIN
        IF "Sales Amount (Actual)" <> 0 THEN
          ProfitPct := ROUND(100 * Profit / "Sales Amount (Actual)",0.1)
        ELSE
          ProfitPct := 0;
      END;
    END;
