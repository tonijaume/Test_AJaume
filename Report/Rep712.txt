  CODE
  {
    VAR
      Text000@1000 : TextConst 'ENU=Period: %1;ESP=Periodo: %1';
      InvPostingGr@1001 : Record 94;
      ItemStatisticsBuf@1007 : Record 5821;
      GLSetup@1013 : Record 98;
      ItemFilter@1002 : Text[250];
      PeriodText@1003 : Text[30];
      SalesQty@1010 : Decimal;
      SalesAmount@1008 : Decimal;
      COGSAmount@1009 : Decimal;
      ItemProfit@1004 : Decimal;
      ItemProfitPct@1005 : Decimal;
      UnitPrice@1011 : Decimal;
      UnitCost@1012 : Decimal;
      PrintAlsoWithoutSale@1006 : Boolean;

    PROCEDURE Calculate@1();
    BEGIN
      SalesQty := -CalcInvoicedQty;
      SalesAmount := CalcSalesAmount;
      COGSAmount := CalcCostAmount + CalcCostAmountNonInvnt;
      ItemProfit := SalesAmount + COGSAmount;

      IF SalesAmount <> 0 THEN
        ItemProfitPct := ROUND(100 * ItemProfit / SalesAmount,0.1)
      ELSE
        ItemProfitPct := 0;

      UnitPrice := CalcPerUnit(SalesAmount,SalesQty);
      UnitCost := -CalcPerUnit(COGSAmount,SalesQty);
    END;

    LOCAL PROCEDURE SetFilters@17();
    VAR
      DimOption@1002 : 'Profit Calculation,Cost Specification,Purch. Item Charge Spec.,Sales Item Charge Spec.,Period,Location';
      DimCodeBuf@1003 : Record 367;
    BEGIN
      WITH ItemStatisticsBuf DO BEGIN
        SETRANGE("Item Filter",Item."No.");
        SETRANGE("Item Ledger Entry Type Filter","Item Ledger Entry Type Filter"::Sale);
        SETFILTER("Entry Type Filter",'<>%1',"Entry Type Filter"::Revaluation);
      END;
    END;

    LOCAL PROCEDURE CalcSalesAmount@12() : Decimal;
    BEGIN
      WITH ItemStatisticsBuf DO BEGIN
        CALCFIELDS("Sales Amount (Actual)");
        EXIT("Sales Amount (Actual)");
      END;
    END;

    LOCAL PROCEDURE CalcCostAmount@5() : Decimal;
    BEGIN
      WITH ItemStatisticsBuf DO BEGIN
        CALCFIELDS("Cost Amount (Actual)");
        EXIT("Cost Amount (Actual)");
      END;
    END;

    LOCAL PROCEDURE CalcCostAmountNonInvnt@15() : Decimal;
    BEGIN
      WITH ItemStatisticsBuf DO BEGIN
        SETRANGE("Item Ledger Entry Type Filter");
        CALCFIELDS("Cost Amount (Non-Invtbl.)");
        EXIT("Cost Amount (Non-Invtbl.)");
      END;
    END;

    LOCAL PROCEDURE CalcInvoicedQty@10() : Decimal;
    BEGIN
      WITH ItemStatisticsBuf DO BEGIN
        SETRANGE("Entry Type Filter");
        CALCFIELDS("Invoiced Quantity");
        EXIT("Invoiced Quantity");
      END;
    END;

    PROCEDURE CalcPerUnit@2(Amount@1000 : Decimal;Qty@1001 : Decimal) : Decimal;
    BEGIN
      IF Qty <> 0 THEN
        EXIT(ROUND(Amount / ABS(Qty),GLSetup."Unit-Amount Rounding Precision"));
      EXIT(0);
    END;

    BEGIN
    END.
