                  SelectedDim@1000 : Record 369;
                BEGIN
                  IF AnalysisViewCode = '' THEN
                    ERROR(Text1140000);

                  IF DateFilter = '' THEN
                    ERROR(Text1140001);

                  DimSelectionBuf.CompareDimText(
                    3,REPORT::"Liquidity Dimensions - Detail",AnalysisViewCode,ColumnDim,Text1140002);

                  TempSelectedDim.RESET;
                  TempSelectedDim.SETFILTER("Dimension Value Filter",'<>%1','');
                  TempSelectedDim.SETFILTER("Dimension Code",TempLiqAcc.TABLECAPTION);
                  IF TempSelectedDim.FIND('-') THEN
                    LiqAcc.SETFILTER("No.",TempSelectedDim."Dimension Value Filter");
                  LiqAcc.SETRANGE("Account Type",LiqAcc."Account Type"::Posting);
                  IF LiqAcc.FIND('-') THEN
                    REPEAT
                      TempLiqAcc.INIT;
                      TempLiqAcc := LiqAcc;
                      TempLiqAcc.INSERT;
                    UNTIL LiqAcc.NEXT = 0;

                  TempLiquidity.INIT;
                  TempLiquidity.INSERT;
                  TempSelectedDim.SETFILTER("Dimension Code",Liquidity.TABLECAPTION);
                  IF TempSelectedDim.FIND('-') THEN
                    Liquidity.SETFILTER("No.",TempSelectedDim."Dimension Value Filter");
                  IF Liquidity.FIND('-') THEN
                    REPEAT
                      TempLiquidity.INIT;
                      TempLiquidity := Liquidity;
                      TempLiquidity.INSERT;
                    UNTIL Liquidity.NEXT = 0;

                  SelectedDim.GetSelectedDim(USERID,3,REPORT::"Liquidity Dimensions - Detail",AnalysisViewCode,TempSelectedDim);
                  TempSelectedDim.RESET;
                  TempSelectedDim.SETCURRENTKEY("User ID","Object Type","Object ID","Analysis View Code",Level);
                  TempSelectedDim.SETFILTER(Level,'<>%1',TempSelectedDim.Level::" ");
                  DimVal.SETRANGE("Dimension Value Type",DimVal."Dimension Value Type"::Standard);
                  IF TempSelectedDim.FIND('-') THEN
                    REPEAT
                      TempDimVal.INIT;
                      TempDimVal.Code := '';
                      TempDimVal."Dimension Code" := TempSelectedDim."Dimension Code";
                      TempDimVal.Name := Text1140003;
                      TempDimVal.INSERT;
                      DimVal.SETRANGE("Dimension Code",TempSelectedDim."Dimension Code");
                      IF TempSelectedDim."Dimension Value Filter" <> '' THEN
                        DimVal.SETFILTER(Code,TempSelectedDim."Dimension Value Filter")
                      ELSE
                        DimVal.SETRANGE(Code);
                      IF DimVal.FIND('-') THEN
                        REPEAT
                          TempDimVal.INIT;
                          TempDimVal := DimVal;
                          TempDimVal.INSERT;
                        UNTIL DimVal.NEXT = 0;
                    UNTIL TempSelectedDim.NEXT = 0;
                END;

    LOCAL PROCEDURE CalcLine@1140000(Level@1140000 : Integer) : Boolean;
    VAR
      HasEntries@1140001 : Boolean;
      i@1140002 : Integer;
    BEGIN
      IF Level < 4 THEN
        FOR i := Level + 1 TO 4 DO
          SetAnaViewEntryFilter(DimCode[i],'*');
      IF Iteration(
        FindFirstDim[Level],DimCode[Level],DimValCode[Level],DimValName[Level],LevelFilter[Level])
      THEN BEGIN
        SetAnaViewEntryFilter(DimCode[Level],DimValCode[Level]);
        HasEntries := AnalysisViewEntry.FIND('-');
      END ELSE
        CurrReport.BREAK;
      EXIT(HasEntries);
    END;

    PROCEDURE PrintDetail@1140001(Level@1140000 : Integer) : Boolean;
    VAR
      AnalysisViewEntryToLiqEntries@1140001 : Codeunit 5055266;
    BEGIN
      IF FindFirstLiqLedgEntry[Level] THEN BEGIN
        FindFirstLiqLedgEntry[Level] := FALSE;
        TempLiqLedgEntry.RESET;
        TempLiqLedgEntry.DELETEALL;
        IF AnalysisViewEntry.FIND('-') THEN BEGIN
          REPEAT
            AnalysisViewEntryToLiqEntries.GetLiqLedgEntries(AnalysisViewEntry,TempLiqLedgEntry);
          UNTIL AnalysisViewEntry.NEXT = 0;
        END;
        TempLiqLedgEntry.SETCURRENTKEY("Liquidity No.","Liquidity Account No.","Source Type","Liquidity Date");
        TempLiqLedgEntry.SETFILTER("Liquidity Date",DateFilter);
        IF NOT TempLiqLedgEntry.FIND('-') THEN
          EXIT(FALSE);
      END ELSE
        IF TempLiqLedgEntry.NEXT = 0 THEN
          EXIT(FALSE);
      IF Level > 1 THEN
        CalcTotalAmounts(Level - 1);
      EXIT(TRUE);
    END;

    LOCAL PROCEDURE CalcTotalAmounts@1140002(Level@1140000 : Integer);
    VAR
      i@1140001 : Integer;
    BEGIN
      FOR i := 1 TO Level DO BEGIN
        Total[i] := Total[i] + TempLiqLedgEntry.Amount;
      END;
    END;

    LOCAL PROCEDURE UpdateColumnDim@1140003();

    LOCAL PROCEDURE Iteration@1140004(VAR FindFirst@1140000 : Boolean;IterationDimCode@1140001 : Text[30];VAR IterationDimValCode@1140002 : Code[20];VAR IterationDimValName@1140003 : Text[50];IterationFilter@1140004 : Text[250]) : Boolean;
    VAR
      SearchResult@1140005 : Boolean;
    BEGIN
      CASE IterationDimCode OF
        TempLiqAcc.TABLECAPTION :
          BEGIN
            TempLiqAcc.RESET;
            TempLiqAcc.SETFILTER("No.",IterationFilter);
            IF FindFirst THEN
              SearchResult := TempLiqAcc.FIND('-')
            ELSE
              IF TempLiqAcc.GET(IterationDimValCode) THEN
                SearchResult := (TempLiqAcc.NEXT <> 0);
            IF SearchResult THEN BEGIN
              IterationDimValCode := TempLiqAcc."No.";
              IterationDimValName := TempLiqAcc.Name;
            END;
          END;
        TempLiquidity.TABLECAPTION :
          BEGIN
            TempLiquidity.RESET;
            TempLiquidity.SETFILTER("No.",IterationFilter);
            IF FindFirst THEN
              SearchResult := TempLiquidity.FIND('-')
            ELSE
              IF TempLiquidity.GET(IterationDimValCode) THEN
                SearchResult := (TempLiquidity.NEXT <> 0);
            IF SearchResult THEN BEGIN
              IterationDimValCode := TempLiquidity."No.";
              IF TempLiquidity."No." <> '' THEN
                IterationDimValName := TempLiquidity.Description
              ELSE
                IterationDimValName := Text1140006;
            END;
          END;
        ELSE BEGIN
          TempDimVal.RESET;
          TempDimVal.SETRANGE("Dimension Code",IterationDimCode);
          TempDimVal.SETFILTER(Code,IterationFilter);
          IF FindFirst THEN
            SearchResult := TempDimVal.FIND('-')
          ELSE
            IF TempDimVal.GET(IterationDimCode,IterationDimValCode) THEN
              SearchResult := (TempDimVal.NEXT <> 0);
          IF SearchResult THEN BEGIN
            IterationDimValCode := TempDimVal.Code;
            IterationDimValName := TempDimVal.Name;
          END;
        END;
      END;
      IF NOT SearchResult THEN BEGIN
        IterationDimValCode := '';
        IterationDimValName := '';
      END;
      FindFirst := FALSE;
      EXIT(SearchResult);
    END;
