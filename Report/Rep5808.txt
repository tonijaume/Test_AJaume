      AverageCostACY@1017 : Decimal;

    PROCEDURE CalcRemainingQty@1();
    BEGIN
      WITH "Item Ledger Entry" DO BEGIN
        FOR i := 1 TO 5 DO BEGIN
          InvtQty[i] := 0;
          IF NOT ISSERVICETIER THEN
            InvtValue[i] := 0;
        END;

        TotalInvtQty := "Remaining Quantity";
        FOR i := 1 TO 5 DO
          IF ("Posting Date" > PeriodStartDate[i]) AND
             ("Posting Date" <= PeriodStartDate[i + 1])
          THEN
            IF "Remaining Quantity" <> 0 THEN BEGIN
              InvtQty[i] := "Remaining Quantity";
              EXIT;
            END;
      END;
    END;

    PROCEDURE CalcUnitCost@2();
    BEGIN
      WITH "Item Ledger Entry" DO BEGIN
        CALCFIELDS("Cost Amount (Expected)","Cost Amount (Actual)");
        UnitCost := ("Cost Amount (Actual)" + "Cost Amount (Expected)") / Quantity;
      END;
    END;

    BEGIN
    END.
