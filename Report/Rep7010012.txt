    BEGIN
      // MontarDatosBanco

      // Rellenar las lineas de wDatosBanco en funcion de lo solicitado por Go Caribic

      IF rBank.IBAN <> '' THEN BEGIN
        wDatosBanco[1] := 'IBAN: ' + rBank.IBAN;

    END;

    }
