          END;
          IF pwReembolso THEN
          BEGIN
            rTMPCobro."Importe Recibido"       -= rCobro."Importe Recibido";
            rTMPCobro."Importe Divisa Reserva" -= rCobro."Importe Divisa Reserva";
          END
          ELSE
          BEGIN
            rTMPCobro."Importe Recibido"       += rCobro."Importe Recibido";
            rTMPCobro."Importe Divisa Reserva" += rCobro."Importe Divisa Reserva";
          END;

        UNTIL rCobro.NEXT = 0;
      END;
    END;
