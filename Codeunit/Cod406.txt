    PROCEDURE UpdateChecklist@7();
    VAR
      SetupChecklistLine@1000 : Record 344;
      AreaLine@1001 : Record 346;
    BEGIN
      WITH GlobalSetupChecklistLine DO BEGIN
        CLEAR(GlobalSetupChecklistLine);
        DELETEALL;
        CLEAR(GlobalAreaLine);
        GlobalAreaLine.DELETEALL;
        MakeChecklist;
        IF SetupChecklistLine.FIND('-') THEN
          REPEAT
            IF GET(SetupChecklistLine."Table ID") THEN BEGIN
              IF (SetupChecklistLine."Line No." <> "Line No.") OR
                 (SetupChecklistLine."Table Name" <> "Table Name") OR
                 (SetupChecklistLine."Form ID" <> "Form ID") OR
                 (SetupChecklistLine."Form Name" <> "Form Name") OR
                 (SetupChecklistLine."License Table ID" <> "License Table ID") OR
                 (SetupChecklistLine."License Table Name" <> "License Table Name") OR
                 (SetupChecklistLine."Copying Available" <> "Copying Available") OR
                 (SetupChecklistLine."Application Area ID" <> "Application Area ID")
              THEN BEGIN
                SetupChecklistLine."Line No." := "Line No.";
                SetupChecklistLine."Table Name" := "Table Name";
                SetupChecklistLine."Form ID" := "Form ID";
                SetupChecklistLine."Form Name" := "Form Name";
                SetupChecklistLine."License Table ID" := "License Table ID";
                SetupChecklistLine."License Table Name" := "License Table Name";
                SetupChecklistLine."Copying Available" := "Copying Available";
                SetupChecklistLine."Application Area ID" := "Application Area ID";
                SetupChecklistLine.MODIFY;
              END
            END ELSE
              SetupChecklistLine.DELETE(TRUE);
          UNTIL SetupChecklistLine.NEXT = 0;
        IF FIND('-') THEN
          REPEAT
            IF NOT SetupChecklistLine.GET("Table ID") THEN BEGIN
              SetupChecklistLine := GlobalSetupChecklistLine;
              SetupChecklistLine.INSERT;
            END;
          UNTIL NEXT = 0;
      END;
      WITH GlobalAreaLine DO BEGIN
        IF AreaLine.FIND('-') THEN
          REPEAT
            IF GET(AreaLine.ID) THEN BEGIN
              IF (AreaLine."Line No." <> "Line No.") OR (AreaLine.Name <> Name) THEN BEGIN
                AreaLine."Line No." := "Line No.";
                AreaLine.Name := Name;
                AreaLine.MODIFY;
              END
            END ELSE
              AreaLine.DELETE(TRUE);
          UNTIL AreaLine.NEXT = 0;
        IF FIND('-') THEN
          REPEAT
            IF NOT AreaLine.GET(ID) THEN BEGIN
              AreaLine := GlobalAreaLine;
              AreaLine.INSERT;
            END;
          UNTIL NEXT = 0;
      END;
    END;

    PROCEDURE CopyDataDialog@3(NewCompanyName@1000 : Text[30];VAR SetupChecklistLine@1001 : Record 344);
    VAR
      ConfirmTableText@1002 : Text[250];
      MessageTableText@1004 : Text[250];
      SingleTable@1003 : Boolean;
    BEGIN
      WITH SetupChecklistLine DO BEGIN
        IF NewCompanyName = '' THEN
          ERROR(Text000);
        IF NOT FIND('-') THEN
          EXIT;
        SingleTable := NEXT = 0;
        IF SingleTable THEN BEGIN
          ConfirmTableText := STRSUBSTNO(Text001,"Table Name",NewCompanyName);
          MessageTableText := STRSUBSTNO(Text002,"Table Name",NewCompanyName);
        END ELSE BEGIN
          ConfirmTableText := STRSUBSTNO(Text003,NewCompanyName);
          MessageTableText := STRSUBSTNO(Text004,NewCompanyName);
        END;
        IF NOT CONFIRM(ConfirmTableText,SingleTable) THEN
          EXIT;
        IF FIND('-') THEN
          REPEAT
            CopyData(SetupChecklistLine);
          UNTIL NEXT = 0;
        COMMIT;
        MESSAGE(MessageTableText)
      END;
    END;

    LOCAL PROCEDURE CopyData@2(VAR SetupChecklistLine@1000 : Record 344);
    VAR
      BaseCompanyName@1001 : Text[30];
    BEGIN
      WITH SetupChecklistLine DO BEGIN
        FILTERGROUP := 2;
        BaseCompanyName := GETRANGEMAX("Company Filter (Source Table)");
        FILTERGROUP := 0;
        IF BaseCompanyName = COMPANYNAME THEN
          ERROR(Text006);
        CALCFIELDS("No. of Records","No. of Records (Source Table)");
        IF "No. of Records" <> 0 THEN
          ERROR(
            Text007 +
            Text008,
            "Table Name",COMPANYNAME);
        IF "No. of Records (Source Table)" = 0 THEN
          ERROR(
            Text009 +
            Text010,
            "Table Name",BaseCompanyName,COMPANYNAME);
        TransferContents("Table ID",BaseCompanyName,TRUE);
      END;
    END;

            IF CopyTable THEN BEGIN
              CHANGECOMPANY(NewCompanyName);
              IF FIND('-') THEN
                REPEAT
                  CHANGECOMPANY(COMPANYNAME);
                  INSERT(TRUE);
                  CHANGECOMPANY(NewCompanyName);
                UNTIL NEXT = 0;
          END;

        ELSE
          EXIT;
      END;
      EXIT(TRUE);
    END;

    LOCAL PROCEDURE MakeChecklistLine@1(TableID@1000 : Integer;FormID@1001 : Integer);
    BEGIN
      WITH GlobalSetupChecklistLine DO BEGIN
        "Table ID" := TableID;
        "Table Name" := ObjTransl.TranslateObject(ObjTransl."Object Type"::Table,"Table ID");
        "License Table ID" := "Table ID";
        "License Table Name" := "Table Name";
        "Form ID" := FormID;
        "Form Name" := ObjTransl.TranslateObject(ObjTransl."Object Type"::Form,"Form ID");;
        "Line No." := "Line No." + 1;
        "Copying Available" := TransferContents(TableID,'',FALSE);
        INSERT;
      END;
    END;

        Name := ModuleName;

      END;
    END;
