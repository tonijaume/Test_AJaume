OBJECT Codeunit 253 BOM Jnl.-B.Post
{
  OBJECT-PROPERTIES
  {
    Date=08/06/01;
    Time=12:00:00;
    Version List=NAVW13.00;
  }
  PROPERTIES
  {
    TableNo=234;
    OnRun=BEGIN
            BOMJnlBatch.COPY(Rec);
            Code;
            Rec.COPY(BOMJnlBatch);
          END;

  }
  CODE
  {
    VAR
      Text000@1000 : TextConst 'ENU=Do you want to post the journals?;ESP=Confirma que desea registrar los diarios?';
      Text001@1001 : TextConst 'ENU=The journals were successfully posted.;ESP=Se han registrado correctamente los diarios.';
      Text002@1002 : TextConst 'ENU="It was not possible to post all of the journals. ";ESP="No ha sido posible registrar todos los diarios. "';
      Text003@1003 : TextConst 'ENU=The journals that were not successfully posted are now marked.;ESP=Los diarios que no se han registrado estn marcados.';
      BOMJnlTemplate@1004 : Record 88;
      BOMJnlBatch@1005 : Record 234;
      BOMJnlLine@1006 : Record 89;
      BOMJnlPostBatch@1007 : Codeunit 33;
      JnlWithErrors@1008 : Boolean;

    LOCAL PROCEDURE Code@1();
    BEGIN
      WITH BOMJnlBatch DO BEGIN
        BOMJnlTemplate.GET("Journal Template Name");
        BOMJnlTemplate.TESTFIELD("Force Posting Report",FALSE);

        IF NOT CONFIRM(Text000,FALSE) THEN
          EXIT;

        FIND('-');
        REPEAT
          BOMJnlLine."Journal Template Name" := "Journal Template Name";
          BOMJnlLine."Journal Batch Name" := Name;
          BOMJnlLine."Line No." := 1;
          CLEAR(BOMJnlPostBatch);
          IF BOMJnlPostBatch.RUN(BOMJnlLine) THEN
            MARK(FALSE)
          ELSE BEGIN
            MARK(TRUE);
            JnlWithErrors := TRUE;
          END;
        UNTIL NEXT = 0;

        IF NOT JnlWithErrors THEN
          MESSAGE(Text001)
        ELSE
          MESSAGE(
            Text002 +
            Text003);

        IF NOT FIND('=><') THEN BEGIN
          RESET;
          FILTERGROUP(2);
          SETRANGE("Journal Template Name","Journal Template Name");
          FILTERGROUP(0);
          Name := '';
        END;
      END;
    END;

    BEGIN
    END.
  }
}
