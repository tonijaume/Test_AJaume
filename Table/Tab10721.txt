    PROCEDURE SetupNewGLAcc@1100000(OldGLAcc@1100000 : Record 10721;BelowOldGLAcc@1100001 : Boolean);
    VAR
      OldGLAcc2@1100002 : Record 10721;
    BEGIN
      IF NOT BelowOldGLAcc THEN BEGIN
        OldGLAcc2 := OldGLAcc;
        OldGLAcc.COPY(Rec);
        OldGLAcc := OldGLAcc2;
        IF NOT OldGLAcc.FIND('<') THEN
          OldGLAcc.INIT;
      END;
      "Income/Balance" := OldGLAcc."Income/Balance";
    END;

    PROCEDURE TranslationMethodConflict@1100004(VAR HistoricGLAcc@1000 : Record 10721) : Boolean;
    BEGIN
      HistoricGLAcc.RESET;
      HistoricGLAcc.SETFILTER("No.",'<>%1',"No.");
      HistoricGLAcc.SETFILTER("Consol. Translation Method",'<>%1',"Consol. Translation Method");
      IF "Consol. Debit Acc." <> '' THEN BEGIN
        IF NOT HistoricGLAcc.SETCURRENTKEY("Consol. Debit Acc.","Consol. Translation Method") THEN
          HistoricGLAcc.SETCURRENTKEY("No.");
        HistoricGLAcc.SETRANGE("Consol. Debit Acc.","Consol. Debit Acc.");
        IF HistoricGLAcc.FIND('-') THEN
          EXIT(TRUE);
        HistoricGLAcc.SETRANGE("Consol. Debit Acc.");
      END;
      IF "Consol. Credit Acc." <> '' THEN BEGIN
        IF NOT HistoricGLAcc.SETCURRENTKEY("Consol. Credit Acc.","Consol. Translation Method") THEN
          HistoricGLAcc.SETCURRENTKEY("No.");
        HistoricGLAcc.SETRANGE("Consol. Credit Acc.","Consol. Credit Acc.");
        IF HistoricGLAcc.FIND('-') THEN
          EXIT(TRUE);
        HistoricGLAcc.SETRANGE("Consol. Credit Acc.");
      END;
      EXIT(FALSE);
    END;

    BEGIN
    END.
