    PROCEDURE DelChangeLogFields@1(InsModDel@1000 : Integer);
    VAR

        REPEAT

            1: ChangeLogSetupField."Log Modification" := FALSE;
            2: ChangeLogSetupField."Log Deletion" := FALSE;

              MODIFY
            ELSE
              DELETE;

    END.
