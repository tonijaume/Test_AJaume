    PROCEDURE fSaldo@1103355000(VAR pwSaldoDL@1100253000 : Decimal) : Decimal;
    VAR
      lrCli@1103355000 : Record 18;
      lrMovCli@1103355002 : Record 21;
      lwSaldo@1103355001 : Decimal;
    BEGIN
      // fSaldo
      // $004

      lwSaldo   := 0;
      pwSaldoDL := 0; //$009

      lrCli.RESET;
      lrCli.SETRANGE("No.", "Customer No.");
      IF GETFILTER("Global Dimension 1 Code") <> '' THEN
        COPYFILTER("Global Dimension 1 Code", lrCli."Global Dimension 1 Code");

      IF GETFILTER("Global Dimension 2 Code") <> '' THEN
        COPYFILTER("Global Dimension 2 Code", lrCli."Global Dimension 2 Code");

      IF lrCli.FIND('-') THEN BEGIN
        lrCli.SETFILTER("Date Filter", '<%1', "Posting Date");
        lrCli.CALCFIELDS("Net Change", "Net Change (LCY)"); //$009

        lwSaldo   := lrCli."Net Change";
        pwSaldoDL := lrCli."Net Change (LCY)";
      END;

      lrMovCli.RESET;
      lrMovCli.SETCURRENTKEY("Customer No.",
                             "Posting Date");
      lrMovCli.COPYFILTERS(Rec);
      lrMovCli.SETRANGE("Customer No.", "Customer No.");
      lrMovCli.SETRANGE("Posting Date", "Posting Date");
      IF lrMovCli.FIND('-') THEN
        REPEAT
          IF lrMovCli."Entry No." <= "Entry No." THEN BEGIN
            lrMovCli.CALCFIELDS(Amount, "Amount (LCY)"); //$009

            lwSaldo   += lrMovCli.Amount;
            pwSaldoDL += lrMovCli."Amount (LCY)"; //$009
          END;
        UNTIL lrMovCli.NEXT = 0;

      EXIT(lwSaldo);
    END;
