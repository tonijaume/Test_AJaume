    PROCEDURE fSaldo@1103355000(VAR pwSaldoDL@1100253000 : Decimal) : Decimal;
    VAR
      lrProv@1103355000 : Record 23;
      lrMovProv@1103355002 : Record 25;
      lwSaldo@1103355001 : Decimal;
    BEGIN
      // fSaldo

      lwSaldo := 0;
      pwSaldoDL := 0; //$001

      lrProv.RESET;
      lrProv.SETRANGE("No.", "Vendor No.");
      IF GETFILTER("Global Dimension 1 Code") <> '' THEN
        COPYFILTER("Global Dimension 1 Code", lrProv."Global Dimension 1 Code");

      IF GETFILTER("Global Dimension 2 Code") <> '' THEN
        COPYFILTER("Global Dimension 2 Code", lrProv."Global Dimension 2 Code");

      IF lrProv.FIND('-') THEN BEGIN
        lrProv.SETFILTER("Date Filter", '<%1', "Posting Date");
        lrProv.CALCFIELDS("Net Change", "Net Change (LCY)"); //$001

        lwSaldo   := (-1) * lrProv."Net Change";
        pwSaldoDL := (-1) * lrProv."Net Change (LCY)"; //$001
      END;

      lrMovProv.RESET;
      lrMovProv.SETCURRENTKEY("Vendor No.",
                              "Posting Date");
      lrMovProv.COPYFILTERS(Rec);
      lrMovProv.SETRANGE("Vendor No.", "Vendor No.");
      lrMovProv.SETRANGE("Posting Date", "Posting Date");
      IF lrMovProv.FINDSET THEN
        REPEAT
          IF lrMovProv."Entry No." <= "Entry No." THEN BEGIN
            lrMovProv.CALCFIELDS(Amount, "Amount (LCY)"); //$001

            lwSaldo := lwSaldo + lrMovProv.Amount;
            pwSaldoDL += lrMovProv."Amount (LCY)"; //$001
          END;
        UNTIL lrMovProv.NEXT = 0;

      EXIT(lwSaldo);
    END;
