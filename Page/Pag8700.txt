      IF MobUserCount > 0 THEN BEGIN
        MobUser.FINDFIRST;

                  More := FALSE;

            SelectionFilter := SelectionFilter + '|';

            SelectionFilter := SelectionFilter + First

          IF MobUserCount > 0 THEN BEGIN
            MobUser.MARKEDONLY(TRUE);
            MobUser.NEXT;
          END;
        END;
      END;
      EXIT(SelectionFilter);
