    BEGIN
      RecRef.OPEN(TableID,TRUE);
      FieldRef := RecRef.FIELD(MigrationColumnField[ColumnID]);

      MigrationData.GET(TableID,"No.",MigrationColumnField[ColumnID]);
      MigrationMgt.CleanFieldError(MigrationData);
      ErrorText := MigrationMgt.EvaluateValue(FieldRef,MatrixCellData[ColumnID]);
      IF (ErrorText <> '') THEN
        ERROR(ErrorText);

      IF FORMAT(FieldRef.TYPE) = 'Option' THEN
        MatrixCellData[ColumnID] := MigrationMgt.GetOptionString(FieldRef,FORMAT(FieldRef.VALUE))

      IF MigrationTableField.Validate AND NOT MigrationMgt.ValidateTableRelation(FieldRef) THEN
        ERROR(STRSUBSTNO(Text001,FieldRef.CAPTION,FieldRef.VALUE));

      MigrationData.VALIDATE(Value,MatrixCellData[ColumnID]);
      MigrationData.MODIFY;
    END;
