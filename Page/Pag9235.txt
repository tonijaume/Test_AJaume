              GLAccAnalysisView.SETFILTER(
                "Dimension 4 Filter",
                AccSchedManagement.GetDimTotalingFilter(4,MatrixRecords[MATRIX_ColumnOrdinal]."Dimension 4 Totaling"));
              GLAccAnalysisView.SETFILTER("Business Unit Filter",MatrixRecords[MATRIX_ColumnOrdinal]."Business Unit Totaling");
              GLAccAnalysisView.FILTERGROUP(0);
              CLEAR(ChartofAccAnalysisView);
              ChartofAccAnalysisView.InsertTempGLAccAnalysisViews(GLAcc);
              ChartofAccAnalysisView.SETTABLEVIEW(GLAccAnalysisView);
              ChartofAccAnalysisView.RUN;
            END;
          END;

          IF (Totaling <> '') AND
             ("Totaling Type" IN ["Totaling Type"::"Cost Type","Totaling Type"::"Cost Type Total"])
          THEN BEGIN
            AccSchedManagement.SetCostTypeRowFilters(CostType,Rec,MatrixRecords[MATRIX_ColumnOrdinal]);
            AccSchedManagement.SetCostTypeColumnFilters(CostType,Rec,MatrixRecords[MATRIX_ColumnOrdinal]);
            WITH MatrixRecords[MATRIX_ColumnOrdinal] DO BEGIN
              CASE "Ledger Entry Type" OF
                "Ledger Entry Type"::"G/L Entries":
                  BEGIN
                    CostTypeChart.SETTABLEVIEW(CostType);
                    CostTypeChart.RUN;
                  END;
                "Ledger Entry Type"::"G/L Budget Entries":
                  BEGIN
                    CostBudPerPeriod.SETTABLEVIEW(CostType);
                    CostBudPerPeriod.RUN;
                  END;
              END;
            END;
          END;
    END;

    LOCAL PROCEDURE MATRIX_OnAfterGetRecord@1079(MATRIX_ColumnOrdinal@1010 : Integer);
    BEGIN
      IF (Totaling = '') OR (MatrixRecords[MATRIX_ColumnOrdinal]."Line No." = 0) THEN
        ColumnValue := 0
      ELSE BEGIN
        ColumnValue := AccSchedManagement.CalcCell(Rec,MatrixRecords[MATRIX_ColumnOrdinal],UseAmtsInAddCurr);
        DivisionError[MATRIX_ColumnOrdinal] := AccSchedManagement.GetDivisionError;
        PeriodError[MATRIX_ColumnOrdinal] := AccSchedManagement.GetPeriodError;
      END;
      MATRIX_CellData[MATRIX_ColumnOrdinal] := ColumnValue;
    END;

    BEGIN
    END.
