                         InvoicedPct[i] := CalcPercentage(UnitPrice[i],ResGrUsagePrice[i][2]);
                         Profit[i] := UnitPrice[i] - ResGrUsageCost[i];
                         ResGrProfitPct[i] := CalcPercentage(Profit[i],UnitPrice[i]);
                       END;

                       SETRANGE("Chargeable Filter");

        EXIT(0);
