          END;
      END;
    END;

      DrillDownForm@1000 : Form 5132;

                                     "Action Taken","Sales Cycle Code","Sales Cycle Stage",

        TableOption::Campaign : OppEntry.SETFILTER("Campaign No.","No.");
        TableOption::Contact : OppEntry.SETFILTER("Contact No.","No.");

      IF GETFILTER("Close Opportunity Filter") <> '' THEN

      IF GETFILTER("Chances of Success % Filter") <> '' THEN
