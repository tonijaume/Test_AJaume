    PROCEDURE CalcMargen@1() : Integer;
    BEGIN
      CALCFIELDS(Ingreso, Coste, "Coste Transfer", "Coste Hoteles");
      IF Ingreso = 0 THEN
        EXIT(0)
      ELSE
        EXIT(ROUND((Ingreso + Coste + "Coste Transfer" + "Coste Hoteles") / Ingreso * 100, 1));
    END;

    BEGIN
    END.
