    PROCEDURE SetFieldList@1(TableID@1001 : Integer;FieldList@1000 : Text[50]);
    VAR
      Field@1002 : Record 2000000041;
      i@1003 : Integer;
      n@1004 : Integer;
    BEGIN
      TempField.DELETEALL;
      SETRANGE(TableNo,TableID);
      IF FieldList = '' THEN
        EXIT;

      i := STRPOS(FieldList,',');
      WHILE i > 0 DO BEGIN
        IF i > 1 THEN BEGIN
          EVALUATE(n,COPYSTR(FieldList,1,i - 1));
          IF Field.GET(TableID,n) THEN BEGIN
            TempField := Field;
            TempField.INSERT;
          END;
        END;
        FieldList := COPYSTR(FieldList,i + 1);
        i := STRPOS(FieldList,',');
      END;
      IF FieldList = '' THEN
        EXIT;

      EVALUATE(n,FieldList);
      IF Field.GET(TableID,n) THEN BEGIN
        TempField := Field;
        TempField.INSERT;
      END;
    END;

    PROCEDURE GetFieldList@9() : Text[50];
    VAR
      Field@1000 : Record 2000000041;
      t@1001 : Text[50];

      IF TempField.FIND('-') THEN

      EXIT(t);
    END;
