  {
    VAR
      MatrixColumnCaptions@1000 : ARRAY [100] OF Text[100];
      MigrationData@1001 : Record 8615;
      MatrixCellData@1002 : ARRAY [100] OF Text[250];
      MigrationColumnField@1004 : ARRAY [100] OF Integer;
      MatrixColumnOrdinal@1003 : Integer;
      FormCaption@1005 : Text[1024];
      TableNo@1006 : Integer;
      Text001@1007 : TextConst 'ENU=%1 value ''%2'' does not exist.;ESP=El valor de %1 ''%2'' no existe.';
      Text002@1008 : TextConst 'ENU=There are no migration data errors in this record.;ESP=No hay errores de datos de migraci¢n en este registro.';
      Text003@1009 : TextConst 'ENU=Data with migration data errors cannot be applied.;ESP=No se pueden aplicar los datos con errores de migraci¢n.';

    PROCEDURE Load@1(VAR NewMatrixColumnCaptions@1002 : ARRAY [100] OF Text[10];FormCaptionIn@1000 : Text[1024];TableNo1@1001 : Integer);
    BEGIN
      COPYARRAY(MatrixColumnCaptions,NewMatrixColumnCaptions,1);
      FormCaption := FormCaptionIn;
      TableNo := TableNo1;
    END;

    PROCEDURE ValidateMigrationData@9(ColumnID@1000 : Integer);
    VAR
      MigrationMgt@1004 : Codeunit 8611;
      RecRef@1003 : RecordRef;
      FieldRef@1002 : FieldRef;
      MigrationTableField@1005 : Record 8616;
      ErrorText@1001 : Text[250];
    BEGIN
      RecRef.OPEN(TableID,TRUE);
      FieldRef := RecRef.FIELD(MigrationColumnField[ColumnID]);

      MigrationData.GET(TableID,"No.",MigrationColumnField[ColumnID]);
      MigrationMgt.CleanFieldError(MigrationData);
      ErrorText := MigrationMgt.EvaluateValue(FieldRef,MatrixCellData[ColumnID]);
      IF (ErrorText <> '') THEN
        ERROR(ErrorText);

      IF FORMAT(FieldRef.TYPE) = 'Option' THEN
        MatrixCellData[ColumnID] := MigrationMgt.GetOptionString(FieldRef,FORMAT(FieldRef.VALUE))
      ELSE
        MatrixCellData[ColumnID] := FORMAT(FieldRef.VALUE);

      MigrationTableField.GET(TableID,MigrationColumnField[ColumnID]);
      IF MigrationTableField.Validate AND NOT MigrationMgt.ValidateTableRelation(FieldRef) THEN
        ERROR(STRSUBSTNO(Text001,FieldRef.CAPTION,FieldRef.VALUE));

      MigrationData.VALIDATE(Value,MatrixCellData[ColumnID]);
      MigrationData.MODIFY;
    END;

    BEGIN
    END.
