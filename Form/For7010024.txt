    PROCEDURE _NombreHotel@1100253002() : Text[250];

      lrCab@1100253000 : Record 7009888;
    BEGIN
      // _NombreHotel

      IF "Nombre Hotel" <> '' THEN
        EXIT("Nombre Hotel");

        EXIT(lrCab."Nombre Hotel");
      END
