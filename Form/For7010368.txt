    PROCEDURE IniciarConductores@6(pGaraje@1100244000 : Code[10];pTA@1100244001 : Code[10];pOrden@1100244002 : 'Garaje+Nombre,Garaje+Tipo Acuerdo+Nombre,Garaje+Tipo Acuerdo+Antiguedad+Nombre,Garaje+Grupo+Tipo Acuerdo+Antiguedad+Nombre');
    VAR
      rCond@1100244003 : Record 7010327;
    BEGIN
      //IniciarConductores.
      //Cargamos TODOS los conductores que procedan en un ARRAY.
      vNumeroConductores := 0;

      rCond.RESET;
      CASE pOrden OF
        0: rCond.SETCURRENTKEY(Garaje,"Nombre Abreviado");
        1: rCond.SETCURRENTKEY(Garaje,"Tipo Acuerdo","Nombre Abreviado");
        2: rCond.SETCURRENTKEY(Garaje,"Tipo Acuerdo","Orden Contrato","Fecha Antiguedad",
                               "Nombre Abreviado");
        3: rCond.SETCURRENTKEY(Garaje,"Grupo Vehiculo","Tipo Acuerdo","Orden Contrato",
                               "Fecha Antiguedad","Nombre Abreviado");
      END;

      IF pGaraje<>'' THEN
        rCond.SETRANGE(Garaje,vGaraje);

      IF pTA <>'' THEN
        rCond.SETRANGE("Tipo Acuerdo",pTA);

      rCond.SETRANGE(Externo,FALSE);

      rCond.SETRANGE(Bloqueado,FALSE);

      IF rCond.FINDSET THEN REPEAT
        vNumeroConductores := vNumeroConductores + 1;
        aTotCond[vNumeroConductores,1] := rCond.Conductor;
        aTotCond[vNumeroConductores,2] := rCond."Nombre Abreviado";
      UNTIL rCond.NEXT=0;

      vPuntero   := 0;
      AvanzaConductores;

      vConductor := aConductores[1,2];
    END;

    PROCEDURE AvanzarHoras@1(pHoras@1100244000 : Integer);
    VAR
      vAux@1100244001 : Integer;
      i@1100244002 : Integer;
      vActIni@1100244003 : Integer;
      vAntIni@1100244004 : Integer;
    BEGIN
      //AvanzarHoras(pHoras);

      EVALUATE(vAntIni,aHoras[1]);

      FOR i:=1 TO CTE_HORAS DO BEGIN
        EVALUATE(vAux,aHoras[i]);
        vAux := vAux+pHoras;

        IF vAux < 0 THEN
          vAux := 24 + vAux
        ELSE
          IF vAux > 23 THEN
            vAux := vAux - 24;

        aHoras[i] := CONVERTSTR(FORMAT(vAux,2),' ','0');
      END;

      //Miramos si debemos modificar la VARIABLE GLOBAL vFecha.
      EVALUATE(vActIni,aHoras[1]);
      IF (pHoras > 0) AND (vActIni < vAntIni) THEN
        vFecha := vFecha + 1
      ELSE
        IF (pHoras < 0) AND (vActIni > vAntIni) THEN
          vFecha := vFecha - 1;

      //...
      DibujarActividad();
    END;

    PROCEDURE AvanzaConductores@13() : Boolean;
    VAR
      rCond@1100244000 : Record 7010327;
      vContador@1100244001 : Integer;
    BEGIN
      //AvanzaConductores;

      IF vPuntero + 1 > vNumeroConductores THEN
        EXIT(FALSE);

      CLEAR(aPuntero);

      vContador := 0;
      REPEAT
        vContador := vContador + 1;

    PROCEDURE DibujarActividad@4();
    VAR
      rP@1100244000 : Record 7010338;
      vNumLinea@1100244001 : Integer;
      vContaPeriodos@1100244002 : Integer;
      vXPos@1100244003 : Integer;
      vAnchura@1100244004 : Integer;
      MENSAJE_ALERTA@1100244005 : TextConst 'ESP=Hay periodos sin validar en alguno de los d¡as visualizados.';
      HABRIA_QUE_PONER_MAS_PERIODOS@1100244006 : TextConst 'ESP=Hay preestablecido un m ximo de 4 periodos por linea y pantalla. Se ha superado ese limite.';
    BEGIN
      //DibujarActividad();

      IniciaGrafico;
      TratamientoDeHoras;

      vAlerta:= ' ';

      rP.RESET;
      rP.SETCURRENTKEY(Conductor,"Fecha Real","Hora Inicio");

      FOR vNumLinea:=1 TO CTE_CONDUCTORES DO BEGIN

        //... Contador de periodos.
        vContaPeriodos  := 0;
        rP.SETRANGE(Conductor,aConductores[vNumLinea,1]);

        //Notese que aprovechamos el valor de la variable vNumeroHoraCero para determinar los
        //d¡as que debemos contemplar.
        //He dejado esta variable global, por si en otro momento se necesitase en otro sitio.
        rP.SETRANGE("Fecha Real",vFecha,vFecha+vNumeroHoraCero);

        IF rP.FINDSET THEN REPEAT
          IF rP.Validado THEN BEGIN

            vContaPeriodos := vContaPeriodos + 1;

            //... Efectuamos la representaci¢n gr fica (si procede por intersecci¢n del periodo con
            //... alguna franja horaria a la vista).
            IF ActividadConductor(rP,vXPos,vAnchura) THEN BEGIN
              IF vContaPeriodos > CTE_MAX_PERIODOS_POR_DIA THEN
                MESSAGE(HABRIA_QUE_PONER_MAS_PERIODOS)
              ELSE
                //... Representamos el periodo por pantalla.
                Representar_Periodo(rP,vXPos,vAnchura,vNumLinea,vContaPeriodos);
            END;

          END
          ELSE BEGIN
            //Anotaremos en una matriz la anomalia encontrada, para m s tarde.
            NotificarAnomalia(aConductores[vNumLinea,1],rP.Fecha,vNumLinea);
            vAlerta := MENSAJE_ALERTA;
          END;

        UNTIL rP.NEXT=0;

        //Analizaremos ahora que todos los registros de actividad del conductor para ese ambito de
        //d¡as. (Basicamente se piensa en la posibilidad que exista un registro de actividad y
        //ning£n periodo creado).
        TestRegistroActividad(aConductores[vNumLinea,1],vFecha,vFecha+vNumeroHoraCero,vNumLinea);

      END;
    END;

    PROCEDURE TestActividadConductor@12(rP@1100244000 : Record 7010338;VAR vFranja@1100244001 : Integer;VAR vHoraIni@1100244002 : Time) : Boolean;
    VAR
      cCond@1100244003 : Codeunit 7010313;
      vHoraFin@1100244004 : Time;
      vTrobat@1100244005 : Boolean;
      vFechaFinPeriodo@1100244006 : Date;
      vAuxFecha@1100244007 : Date;
    BEGIN
      //TestActividadConductor(rP,@vFranja,@vHoraIni)=>TRUE si se ha encontrado actividad en la franja.
      //                                               FALSE en caso contrario.

      vAuxFecha:= vFecha;

      //... Buscamos la fecha en la que acaba el periodo.

      IF rP."Hora Fin" >= rP."Hora Inicio" THEN
        vFechaFinPeriodo := rP."Fecha Real"
      ELSE
        vFechaFinPeriodo := rP."Fecha Real" + 1;

      vFranja:=1;  //Contador de franjas horarias.
      vTrobat := FALSE;

      WHILE (vFranja<=CTE_HORAS) AND (vTrobat =FALSE) DO BEGIN

        EVALUATE(vHoraIni,aHoras[vFranja]);

        vHoraFin := (vHoraIni + CTE_FACTOR_HORA) - 60000; //Descontamos un minuto.

        //... Miramos a que fecha corresponde la franja que estamos analizando.

        IF (aHoras[vFranja] = '00') AND (vFranja > 1) THEN BEGIN
          vAuxFecha   := vAuxFecha + 1;
        END;

        //... Con las variables adecuadas ya calculadas, miramos si hay interseccion del periodo
        //... con la franja horaria.
        IF cCond.InterseccionHoras(vAuxFecha,vAuxFecha,rP."Fecha Real",vFechaFinPeriodo,
                                   vHoraIni,vHoraFin,rP."Hora Inicio",rP."Hora Fin") THEN
          vTrobat := TRUE
        ELSE
          vFranja := vFranja+1;

      END;

      EXIT(vTrobat);
    END;

    PROCEDURE ActividadConductor@5(rP@1100244000 : Record 7010338;VAR vXPos@1100244001 : Integer;VAR vAnchura@1100244002 : Integer) : Boolean;
    VAR
      vDuracion@1100244003 : Decimal;
      vHoraIni@1100244004 : Time;
      vFranja@1100244005 : Integer;
      vReferencia@1100244006 : Time;
    BEGIN
      //ActividadConductor(rP,@vXPos,@vAnchura);
      //Buscamos, si procede, la posicion X y la anchura que debe tener el periodo en la pantalla.

      //... Miramos en primer lugar si encontramos para ese periodo alguna franja horaria de la
      //... pantalla con la que intersecte.

      IF NOT TestActividadConductor(rP,vFranja,vHoraIni) THEN
        EXIT(FALSE);

      //... La variable VFRANJA nos dice la primera franja horaria donde se ha encontrado actividad.
      //... La variable VHORAINI nos dice la hora en que empieza la franja.

      vXPos    := 0;
      vAnchura := 0;

      //... Miramos si como inicio (para la representaci¢n) consideramos el inicio del periodo o bien
      //... la hora de inicio de la primera franja.

      vReferencia := Maximo(rP."Fecha Real",rP."Hora Inicio",vFecha,vHoraIni);
      vDuracion := ROUND( CTE_ANCHURA * (vReferencia - vHoraIni) / CTE_FACTOR_HORA,1);
      vXPos     := CTE_POSX + (CTE_ANCHURA * (vFranja-1)) + vDuracion;

      //...Calculamos la duraci¢n del periodo.

      IF rP."Hora Fin" >= vReferencia THEN
        vDuracion := (rP."Hora Fin" - vReferencia) / CTE_FACTOR_HORA
      ELSE
        vDuracion := ( (rP."Hora Fin" - 000000T) + (235959T - vReferencia) ) / CTE_FACTOR_HORA;

      // ... Calculamos por otra parte, la longitud que debe tener.

      vAnchura  := ROUND(vDuracion * CTE_ANCHURA,1);

      // ... Tenemos en cuenta que nos pasemos por la parte derecha.
      // ... Si nos pasamos, efectuamos la correcci¢n.

      IF vXPos + vAnchura > CTE_LIMITE_DERECHA THEN
        vAnchura := CTE_LIMITE_DERECHA - vXPos;

      //Se ha podido encontrar intersecci¢n (s¢lo 1 minuto), ==> que la anchura es cero.
      //Esta situaci¢n anomala no debe representarse por pantalla.

      IF vAnchura > 0 THEN
        EXIT(TRUE)
      ELSE
        EXIT(FALSE);
    END;

    PROCEDURE NotificarAnomalia@21(pConductor@1100244000 : Code[10];pFecha@1100244001 : Date;pNumLinea@1100244002 : Integer);
    BEGIN
      //NotificarAnomalia(pConductor)
      IF aAnomalia[pNumLinea] = '' THEN BEGIN
        aAnomalia[pNumLinea]    := pConductor;
        aAnomalia2[pNumLinea,1] :=pFecha;
        aAnomalia2[pNumLinea,2] :=pFecha;
        //Aprovecharemos para resaltar el error por pantalla.
        VisualizarAnomalia(pNumLinea);
      END
      ELSE BEGIN
        IF pFecha < aAnomalia2[pNumLinea,1] THEN
          aAnomalia2[pNumLinea,1] := pFecha
        ELSE BEGIN
          IF pFecha > aAnomalia2[pNumLinea,2] THEN
            aAnomalia2[pNumLinea,2] := pFecha;
        END;
      END;
    END;

    PROCEDURE GestionConductor@2(pConductor@1100244000 : Code[10];pNumLinea@1100244001 : Integer);
