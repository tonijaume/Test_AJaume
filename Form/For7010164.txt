    BEGIN
      IF "Tipo aplicacion" = "Tipo aplicacion"::Pax THEN BEGIN
        VALIDATE(Precio, 0);
        CurrForm.Precio.EDITABLE          := FALSE;
        CurrForm."Precio Adulto".EDITABLE := TRUE;
